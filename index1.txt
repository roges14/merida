  <script>
    function switchMain(option) {
      document.querySelectorAll('.option-panel').forEach(p => p.classList.remove('active'));
      document.getElementById('panel-' + option).classList.add('active');
      document.querySelectorAll('.tab-btn').forEach((b, i) => {
        b.classList.toggle('active', (i === 0 && option === 'celestun') || (i === 1 && option === 'coloradas'));
      });
    }

    function switchPriv(option) {
      ['ppanel-pcele', 'ppanel-pcolo'].forEach(id => document.getElementById(id).classList.remove('active'));
      document.getElementById('ppanel-' + option).classList.add('active');
      ['ptab-celestun', 'ptab-coloradas'].forEach((id, i) => {
        document.getElementById(id).classList.toggle('active', (i === 0 && option === 'pcele') || (i === 1 && option === 'pcolo'));
      });
    }

    function switchCard(btn, paneId) {
      const card = btn.closest('.card');
      card.querySelectorAll('.card-tab').forEach(t => t.classList.remove('active'));
      card.querySelectorAll('.card-pane').forEach(p => p.classList.remove('active'));
      btn.classList.add('active');
      document.getElementById(paneId).classList.add('active');
    }
  </script>
